/**
 * ===================== require vs import (Node.js Notes) =====================
 *
 * ðŸ”¹ CommonJS (require)
 * ---------------------------------------------------------------------------
 * - require() ek normal function hota hai
 * - Code line-by-line (top to bottom) execute hota hai
 * - Jahan require() likha hota hai, wahi pe:
 *     â†’ module load hota hai
 *     â†’ module ka top-level code execute hota hai
 * - Node.js require() ko koi extra priority nahi deta
 *
 * Example:
 *   console.log(1);
 *   const data = require("./file.js");
 *   console.log(2);
 *
 * Output:
 *   1
 *   (file.js ka output)
 *   2
 *
 * ðŸ‘‰ Isliye require() use karne par execution pure sequence me hota hai
 *
 *
 * ðŸ”¹ ES Modules (import)
 * ---------------------------------------------------------------------------
 * - import ek special syntax hota hai (normal function nahi)
 * - import statements hoisted hote hain
 * - File ka code run hone se pehle:
 *     â†’ saare imports resolve hote hain
 *     â†’ imported files execute hoti hain
 * - Isliye Node.js import ko priority deta hai
 *
 * Example:
 *   console.log(1);
 *   import "./file.js";
 *   console.log(2);
 *
 * Output:
 *   (file.js ka output)
 *   1
 *   2
 *
 *
 * ðŸ”¹ Main Difference (One Line)
 * ---------------------------------------------------------------------------
 * require()  â†’ runtime execution (no priority, sequence follow hota hai)
 * import     â†’ compile-time execution (pehle import, baad main code)
 *
 *
 * ðŸ”¹ Golden Rule (Important)
 * ---------------------------------------------------------------------------
 * import  â†” export
 * require â†” module.exports
 *
 * âŒ Dono system ko mix nahi karna chahiye
 * âœ… Ek project me ek hi module system use karo
 *
 * ---------------------------------------------------------------------------
 * "require is executed at runtime line by line,
 *  while import statements are hoisted and executed before the main code."
 *
 * ============================================================================
 */

/**
 * const data = await User.find();
 *
 * Step by step explanation:
 *
 * 1ï¸âƒ£ User
 * - 'User' ek Mongoose model hai.
 * - Ye model MongoDB ke collection se link hota hai.
 *   Example: Agar tumne schema define kiya hai 'UserSchema' aur model banaya hai 'User',
 *   toh MongoDB me automatically ek collection create hota hai jiska naam by default 'users' hota hai.
 * - Important: 'User' model collection ka reference hai, collection khud ye variable nahi hai.
 *
 * 2ï¸âƒ£ .find()
 * - '.find()' Mongoose ka method hai jo collection me se documents fetch karta hai.
 * - Agar tum isko bina argument ke call karte ho (User.find()), iska matlab hai:
 *     "collection me se sab documents fetch karo"
 * - Agar tum argument pass karte (User.find({age: 20})), iska matlab hai filter laga ke fetch karna.
 *
 * 3ï¸âƒ£ await
 * - '.find()' ek asynchronous operation hai, kyunki ye database se data fetch karta hai.
 * - 'await' ka matlab: "jab tak database se result na aa jaye, next line pe mat jao"
 * - Iske bina tumhe .then() ka use karna padta.
 *
 * 4ï¸âƒ£ const data = ...
 * - Ye variable me result store karta hai.
 * - data ka type: Array
 *   - Is array ke andar documents (objects) hote hain jo MongoDB collection me stored hain.
 *   - Matlab: data me ek single document nahi, balki sab documents ka array hota hai.
 *
 * 5ï¸âƒ£ Example
 * Maan lo collection me ye 2 documents hain:
 * [
 *   { "_id": "1", "name": "Vivek", "email": "vivek@gmail.com" },
 *   { "_id": "2", "name": "Sohan", "email": "sohan@gmail.com" }
 * ]
 *
 * const data = await User.find();
 * console.log(data);
 *
 * Output:
 * [
 *   { _id: "1", name: "Vivek", email: "vivek@gmail.com" },
 *   { _id: "2", name: "Sohan", email: "sohan@gmail.com" }
 * ]
 *
 * - Dhyan do, ye array hai, aur array ke andar sab documents objects ke form me hain.
 *
 * âœ… Summary
 * - User â†’ Mongoose model, collection ka reference.
 * - .find() â†’ collection me se documents fetch karna.
 * - await â†’ asynchronous wait karna.
 * - data â†’ array of documents (sab documents ka collection nahi, collection ka data hai).
 */

/**
 * const user = await User.findById(req.params.id);
 *
 * Step by step explanation:
 *
 * 1ï¸âƒ£ User
 * - 'User' ek Mongoose model hai.
 * - Ye MongoDB ke 'users' collection se linked hai.
 * - Collection me sab documents store hote hain (jaise pehle 'find()' me dekha).
 *
 * 2ï¸âƒ£ .findById()
 * - Ye Mongoose ka method hai.
 * - Ye specific document ko find karta hai uske '_id' ke basis par.
 * - Argument me document ka _id dena padta hai.
 *   Example: User.findById("64f0e3a2c5a3") â†’ jo document ka _id ye hai, wo milega.
 * - Difference from .find():
 *   - .find() â†’ sab documents ka array return karta hai.
 *   - .findById() â†’ ek single document return karta hai (ya null agar document exist nahi karta).
 *
 * 3ï¸âƒ£ req.params.id
 * - Ye Express.js ka object hai.
 * - req.params me URL parameters store hote hain.
 * - Agar route hai '/users/:id' aur user ne URL me '/users/64f0e3a2c5a3' hit kiya:
 *     req.params.id = "64f0e3a2c5a3"
 * - Ye findById() ke argument ke liye use hota hai.
 *
 * 4ï¸âƒ£ await
 * - .findById() bhi asynchronous operation hai.
 * - 'await' ka matlab: "Jab tak database se result na aa jaye, next line pe mat jao"
 * - Iske bina .then() ka use karna padta.
 *
 * 5ï¸âƒ£ const user = ...
 * - 'user' variable me single document store hota hai.
 * - Agar document exist karta hai â†’ object milega:
 *   {
 *     _id: "64f0e3a2c5a3",
 *     name: "Vivek",
 *     email: "vivek@gmail.com"
 *   }
 * - Agar document exist nahi karta â†’ null return hota hai.
 *
 * 6ï¸âƒ£ Example
 * Route:
 * app.get("/users/:id", async (req, res) => {
 *   const user = await User.findById(req.params.id);
 *   console.log(user);
 * });
 *
 * URL hit: /users/64f0e3a2c5a3
 *
 * Output (agar document exist karta hai):
 * {
 *   _id: "64f0e3a2c5a3",
 *   name: "Vivek",
 *   email: "vivek@gmail.com"
 * }
 *
 * Agar document exist nahi karta â†’ output:
 * null
 *
 * âœ… Summary
 * - User â†’ Mongoose model (collection ka reference).
 * - .findById() â†’ ek specific document fetch karta hai _id ke basis par.
 * - req.params.id â†’ URL se aaya hua id.
 * - await â†’ asynchronous wait.
 * - user â†’ single document object ya null.
 */
